            if (termoBusca) {

                dadosFiltrados = dadosFiltrados.filter(a => a.nomePopular.toLowerCase().includes(termoBusca.toLowerCase()));

            }

            if (filtroAtivo !== "todos") {

                dadosFiltrados = dadosFiltrados.filter(a => a.atributos[filtroAtivo] === 5);

            }

            renderCards(dadosFiltrados);

        }


        inputBusca.addEventListener('input', (e) => {

            termoBusca = e.target.value;

            aplicarFiltros();

        });


        botoesFiltro.forEach(botao => {

            botao.addEventListener('click', (e) => {

                botoesFiltro.forEach(b => b.classList.remove('active'));

                e.target.classList.add('active');

                filtroAtivo = e.target.getAttribute('data-filter');

                aplicarFiltros();

            });

        });


        // --- 5. INICIALIZAÇÃO ---

        renderCards(arbores);
